109 def _generate_number(self, prefix, length): 

110 """ 

111 'prefix' is the start of the CC number as a string, any number of digits. 

112 'length' is the length of the CC number to generate. Typically 13 or 16 

113 """ 

114 number = prefix 

115 # Generate random char digits 

116 number += '#' * (length - len(prefix) - 1) 

117 number = self.numerify(number) 

118 reverse = number[::-1] 

119 # Calculate sum 

120 tot = 0 

121 pos = 0 

122 while pos < length - 1: 

123 tot += Provider.luhn_lookup[reverse[pos]] 

124 if pos != (length - 2): 

125 tot += int(reverse[pos + 1]) 

126 pos += 2 

127 # Calculate check digit 

128 check_digit = (10 - (tot % 10)) % 10 

129 number += str(check_digit) 

130 return number
