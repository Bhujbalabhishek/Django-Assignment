50 def get_environ(self, request): 

51 """ 

52 Given a `requests.PreparedRequest` instance, return a WSGI environ dict. 

53 """ 

54 method = request.method 

55 url = request.url 

56 kwargs = {} 

57 

58 # Set request content, if any exists. 

59 if request.body is not None: 

60 if hasattr(request.body, 'read'): 

61 kwargs['data'] = request.body.read() 

62 else: 

63 kwargs['data'] = request.body 

64 if 'content-type' in request.headers: 

65 kwargs['content_type'] = request.headers['content-type'] 

66 

67 # Set request headers. 

68 for key, value in request.headers.items(): 

69 key = key.upper() 

70 if key in ('CONNECTION', 'CONTENT-LENGTH', 'CONTENT-TYPE'): 

71 continue 

72 kwargs['HTTP_%s' % key.replace('-', '_')] = value 

73 

74 return self.factory.generic(method, url, **kwargs).environ

76 def send(self, request, *args, **kwargs): 

77 """ 

78 Make an outgoing request to the Django WSGI application. 

79 """ 

80 raw_kwargs = {} 

81 

82 def start_response(wsgi_status, wsgi_headers): 

83 status, _, reason = wsgi_status.partition(' ') 

84 raw_kwargs['status'] = int(status) 

85 raw_kwargs['reason'] = reason 

86 raw_kwargs['headers'] = wsgi_headers 

87 raw_kwargs['version'] = 11 

88 raw_kwargs['preload_content'] = False 

89 raw_kwargs['original_response'] = MockOriginalResponse(wsgi_headers) 

90 

91 # Make the outgoing request via WSGI. 

92 environ = self.get_environ(request) 

93 wsgi_response = self.app(environ, start_response) 

94 

95 # Build the underlying urllib3.HTTPResponse 

96 raw_kwargs['body'] = io.BytesIO(b''.join(wsgi_response)) 

97 raw = requests.packages.urllib3.HTTPResponse(**raw_kwargs) 

98 

99 # Build the requests.Response 

100 return self.build_response(request, raw)

149 def _encode_data(self, data, format=None, content_type=None): 

150 """ 

151 Encode the data returning a two tuple of (bytes, content_type) 

152 """ 

153 

154 if data is None: 

155 return ('', content_type) 

156 

157 assert format is None or content_type is None, ( 

158 'You may not set both `format` and `content_type`.' 

159 ) 

160 

161 if content_type: 

162 # Content type specified explicitly, treat data as a raw bytestring 

163 ret = force_bytes(data, settings.DEFAULT_CHARSET) 

164 

165 else: 

166 format = format or self.default_format 

167 

168 assert format in self.renderer_classes, ( 

169 "Invalid format '{}'. Available formats are {}. " 

170 "Set TEST_REQUEST_RENDERER_CLASSES to enable " 

171 "extra request formats.".format( 

172 format, 

173 ', '.join(["'" + fmt + "'" for fmt in self.renderer_classes]) 

174 ) 

175 ) 

176 

177 # Use format and render the data into a bytestring 

178 renderer = self.renderer_classes[format]() 

179 ret = renderer.render(data) 

180 

181 # Determine the content-type header from the renderer 

182 content_type = renderer.media_type 

183 if renderer.charset: 

184 content_type = "{}; charset={}".format( 

185 content_type, renderer.charset 

186 ) 

187 

188 # Coerce text to bytes if required. 

189 if isinstance(ret, str): 

190 ret = ret.encode(renderer.charset) 

191 

192 return ret, content_type

360class URLPatternsTestCase(testcases.SimpleTestCase): 

361 """

364 class ATestCase(URLPatternsTestCase):

383 cls._module_urlpatterns = cls._module.urlpatterns

385 cls._module.urlpatterns = cls.urlpatterns

387 cls._override.enable() 

388 super().setUpClass()

392 super().tearDownClass() 

393 cls._override.disable()
